-- EJEMPLO
/*
Consulta 1: Muestra el nombre completo de los empleados (FirstName y LastName) y el total de pedidos gestionados por cada uno.
Solo incluye empleados que han gestionado más de 20 pedidos, ordenados por el total de pedidos en orden descendente.
*/

SELECT 
    E.FirstName + ' ' + E.LastName AS NombreCompleto, 
    COUNT(O.OrderID) AS TotalPedidos 
FROM 
    Employees E
JOIN 
    Orders O 
    ON E.EmployeeID = O.EmployeeID 
GROUP BY 
    E.FirstName, 
    E.LastName 
HAVING 
    COUNT(O.OrderID) > 20 
ORDER BY 
    TotalPedidos DESC; 

/*
Consulta 2: Muestra los nombres de los productos (ProductName) cuyo precio (UnitPrice) es mayor al precio promedio de todos los productos.
*/
SELECT 
    ProductName, 
    UnitPrice
FROM 
    Products
WHERE 
    UnitPrice > ( -- Subconsulta: Calcula el precio promedio de todos los productos.
        SELECT 
            AVG(UnitPrice)
        FROM 
            Products
    )
ORDER BY 
    UnitPrice DESC;
